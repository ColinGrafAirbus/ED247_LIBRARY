179 static int gettimeofday(struct timeval * tp, struct timezone * tzp)
180 {
181  // Note: some broken versions only have 8 trailing zero's, the correct epoch has 9 trailing zero's
182  // This magic number is the number of 100 nanosecond intervals since January 1, 1601 (UTC)
183  // until 00:00:00 January 1, 1970
184  static const uint64_t EPOCH = ((uint64_t) 116444736000000000ULL);
185 
186  SYSTEMTIME system_time;
187  FILETIME file_time;
188  uint64_t time;
189 
190  GetSystemTime( &system_time );
191  SystemTimeToFileTime( &system_time, &file_time );
192  time = ((uint64_t)file_time.dwLowDateTime ) ;
193  time += ((uint64_t)file_time.dwHighDateTime) << 32;
194 
195  tp->tv_sec = (long) ((time - EPOCH) / 10000000L);
196  tp->tv_usec = (long) (system_time.wMilliseconds * 1000);
197  return 0;
198 }
