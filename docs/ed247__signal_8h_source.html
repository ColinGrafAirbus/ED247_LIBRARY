97  static size_t sample_max_size_bytes(const ed247_signal_info_t & info)
98  {
99  switch(info.type){
100  case ED247_SIGNAL_TYPE_DISCRETE:
101  return 1;
102  case ED247_SIGNAL_TYPE_ANALOG:
103  return 4;
104  case ED247_SIGNAL_TYPE_NAD:
105  {
106  size_t size = 1;
107  for(unsigned i = 0 ; i < info.info.nad.dimensions_count ; i++){
108  size *= info.info.nad.dimensions[i];
109  }
110  size = size*xml::nad_type_size(info.info.nad.nad_type);
111  return size;
112  }
113  case ED247_SIGNAL_TYPE_VNAD:
114  return xml::nad_type_size(info.info.vnad.nad_type) * info.info.vnad.max_length;
115  default:
116  return 0;
117  }
118  }
